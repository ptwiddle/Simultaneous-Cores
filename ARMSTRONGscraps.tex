\begin{lemma}
$$\sum_{i\in I_a} i^2=\frac{a^3-a}{12}$$
\end{lemma}

\begin{proof}
\begin{align*}
\sum_{i\in I_a} i^2&=\sum_{j=0}^{a-1} (j-\frac{a-1}{2})^2 \\
&=\sum_{j=0}^{a-1} j^2-(a-1)\sum_{j=0}^{a-1}j+a\frac{(a-1)^2}{4} \\
&=\frac{(a-1)a(2a-1)}{6}-(a-1)a(a-1)/2+a(a-1)^2/4 \\
&=\frac{a(a-1)}{12}(4a-2-6(a-1)+3(a-1)) \\
&=\frac{a^3-a}{12}
\end{align*}
\end{proof}
